20static void tune_uclamp_min() {
21 int32_t properties = 0;
22
23 properties = SET_REQUEST_PRIORITY(properties, REQ_PRIORITY_LOW);
24 properties = ADD_ALLOWED_MODE(properties, MODE_SUSPEND);
25
26 SysResource resourceList[] = {
27 {
28 .mResCode = UCLAMP_MIN_RES,
29 .mResInfo = 0,
30 .mOptionalInfo = 0,
31 .mNumValues = 1,
32 .mResValue = {
33 .value = UCLAMP_BOOST_VAL,
34 }
35 }
36 };
37
38 int64_t handle = tuneResources(TUNE_DURATION, properties, 1, resourceList);
39 if(handle == FAIL) {
40 printf("Request Could not be Sent to the Resource Tuner Server\n");
41 } else {
42 printf("Handle Returned is: %ld\n", handle);
43 }
44}
